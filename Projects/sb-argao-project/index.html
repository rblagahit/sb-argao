        function showToast(msg) {

        window.navigateTo = (view) => {

        window.handleNavAction = () => {

        const startSync = () => {

            if (unsubDocs) unsubDocs();

            if (unsubMembers) unsubMembers();


            unsubMembers = onSnapshot(MEMBER_REF, (snap) => {

                localMembers = snap.docs.map(d => ({id: d.id, ...d.data()}));

                renderUI();

            }, (err) => console.warn("Member sync issue:", err));


            unsubDocs = onSnapshot(DOC_REF, (snap) => {

                localDocs = snap.docs.map(d => ({id: d.id, ...d.data()}));

                renderUI();

            }, (err) => console.warn("Doc sync issue:", err));

        };


        function renderUI() {

            const mRow = document.getElementById('memberRow');

            if (mRow) {

                if (localMembers.length === 0) {

                   mRow.innerHTML = `<p class="text-xs text-slate-400 font-bold uppercase tracking-widest">Public Archive Mode</p>`;

                } else {

                    mRow.innerHTML = localMembers.map(m => `

                        <div class="flex items-center gap-3 bg-white border border-slate-100 py-3 px-5 rounded-2xl cursor-pointer hover:border-blue-200 transition-all flex-shrink-0" onclick="window.viewMember('${m.id}')">

                            <img src="${m.image}" class="w-8 h-8 rounded-lg object-cover" onerror="this.src='https://ui-avatars.com/api/?name=${encodeURIComponent(m.name)}'">

                            <span class="text-[11px] font-black text-slate-800">${m.name}</span>

                        </div>

                    `).join('');

                }

            }

            renderDocs();

        }


        function renderDocs() {

            const grid = document.getElementById('documentGrid');

            if (!grid) return;

            const term = document.getElementById('searchInput').value.toLowerCase();

            const typeFilter = document.getElementById('typeFilter').value;


            const filtered = localDocs.filter(d => {

                return (d.title.toLowerCase().includes(term) || d.docId.toLowerCase().includes(term)) &&

                       (typeFilter === 'All' || d.type === typeFilter);

            });


            if (filtered.length === 0) {

                grid.innerHTML = `<div class="col-span-full py-20 text-center text-slate-400 font-bold uppercase tracking-widest">No documents found</div>`;

                return;

            }


            grid.innerHTML = filtered.map(doc => {

                return `

                    <div class="doc-card">

                        <div class="p-10 border-b border-slate-50">

                            <span class="text-[10px] font-black px-4 py-1.5 rounded-full uppercase tracking-widest bg-blue-600 text-white">${doc.type}</span>

                            <h3 class="text-2xl font-extrabold text-slate-900 mt-6 mb-8 leading-[1.3]">${doc.title}</h3>

                            <div class="flex items-center gap-4">

                                <img src="${author.image}" class="w-12 h-12 rounded-xl object-cover bg-slate-100" onerror="this.src='https://upload.wikimedia.org/wikipedia/commons/7/7e/Ph_seal_cebu_argao.png'">

                                <div>

                                    <p class="text-xs font-black text-slate-900">${author.name}</p>

                                    <p class="text-[9px] font-bold text-slate-400 uppercase">Sponsor</p>

                            </div>

                        <div class="p-10">

                            <a href="${doc.link}" target="_blank" class="w-full bg-slate-900 text-white flex items-center justify-center gap-3 py-4 rounded-2xl font-black uppercase text-[11px] hover:bg-blue-600 transition-all">

                                View PDF <i class="far fa-file-pdf"></i>

                            </a>

                        </div>

                    </div>

                `;

            }).join('');

        }
